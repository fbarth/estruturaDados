\begin{Slide}{Introdução - Pilha}
  \begin{itemize}
  \item novo elemento é inserido no topo e acesso é apenas ao topo
    \begin{itemize}
      \item o primeiro que sai é o último que entrou (LIFO -
        \textit{last in, first out}) 
    \end{itemize}
   \item operações básicas:

\begin{itemize}
\item empilhar (\textit{push}) um novo elemento, inserindo-o no topo
\item desempilhar (\textit{pop}) um elemento, removendo-o do topo
\end{itemize}
   
  \end{itemize}
